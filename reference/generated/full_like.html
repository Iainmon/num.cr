def full_like(other : NDTensor, x : Number, dtype : U.class = Float64) forall U

Initializes a Tensor filled with the provided value, whose size is inferred from a given Tensor

t = Tensor.new [1, 2, 3]

f = full_like(t, -1, dtype: Int32)
f # => Tensor[-1, -1, -1]
